\begin{frame}{Facebook Webhook: Luồng Tiếp Nhận Tin}
    \begin{center}
        \large\textbf{\color{accent}Bắt sự kiện tương tác thời gian thực từ Fanpage.}
    \end{center}
    \vspace{0.4cm}
    \begin{block}{Quy trình xử lý an toàn}
        \begin{itemize}
            \item \textbf{Đường truyền}: Mở cổng HTTPS công khai qua Ngrok.
            \item \textbf{Xác thực}: Kiểm tra chữ ký HMAC-SHA256 bảo mật.
            \item \textbf{Messenger}: Trích xuất nội dung tin nhắn và PSID.
            \item \textbf{Comments}: Lọc bình luận, bỏ qua echo của Page.
            \item \textbf{Đầu ra}: Đẩy sự kiện chuẩn hóa vào Kafka.
        \end{itemize}
    \end{block}
\end{frame}

% SLIDE 6: Facebook Outbound Flow (Tích hợp FB - Luồng ra)

\begin{frame}{Facebook Graph API: Luồng Phản Hồi Tin}
    \begin{center}
        \large\textbf{\color{accent}Tự động phản hồi khách hàng qua Graph API.}
    \end{center}
    \vspace{0.4cm}
    \begin{block}{Các bước thực thi}
        \begin{itemize}
            \item \textbf{Token}: Dùng Page Access Token vĩnh viễn mới.
            \item \textbf{Messenger API}: Gửi tin nhắn trực tiếp theo PSID.
            \item \textbf{Comment API}: Trả lời bình luận theo Comment ID.
            \item \textbf{Phân luồng}: Đảm bảo tin nhắn đi đúng chuỗi.
            \item \textbf{An toàn}: Ghi nhật ký trạng thái gửi (SENT/FAILED).
        \end{itemize}
    \end{block}
\end{frame}

            \item \textbf{Trùng lặp}: Dựa trên Message-ID để lọc thư trùng.
            \item \textbf{Xử lý}: Làm sạch mã HTML, lọc văn bản thô.
            \item \textbf{Lưu trữ}: Nhận diện và tách riêng tệp đính kèm.

% SLIDE 8: Email Outbound Flow (Tích hợp Email - Luồng ra)
% -------------------------------------------------------------
\begin{frame}{Email SMTP: Luồng Gửi Thư Phản Hồi}
    \begin{center}
        \large\textbf{\color{accent}Bảo toàn luồng chuỗi thư trao đổi (Threaded Mail).}
    \end{center}
    \vspace{0.4cm}
    \begin{block}{Bảo toàn chuỗi hội thoại}
        \begin{itemize}
            \item \textbf{SMTP}: Sử dụng Spring `JavaMailSender` gửi thư đi.
            \item \textbf{In-Reply-To}: Gắn ID của email gửi đến gần nhất.
            \item \textbf{References}: Gắn danh sách ID của cả luồng thư.
            \item \textbf{Hiển thị}: Gom nhóm chuỗi thư chuẩn trên Gmail/Outlook.
            \item \textbf{Giả lập}: Test offline không cần cấu hình bằng Mailpit.
        \end{itemize}
    \end{block}
\end{frame}
